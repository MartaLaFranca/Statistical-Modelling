Si carica la funzione white.test che verrà utilizzata durante lo studio dell’eteroschedasticità:

white.test<-function(lmod){
  u2<-lmod$residuals^2
  y<-lmod$fitted
  R2u<-summary(lm(u2~y+I(y^2)))$r.squared
  LM<-length(y)*R2u
  p.val<-1-pchisq(LM,2)
  data.frame("Test Statistic"=LM, "P"=p.val)
}
